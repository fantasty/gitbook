# 反转链表

```cpp

LinklistNode* ReverseList(LinklistNode* head)
{
    if (head == NULL || head->m_Next == NULL)
        return head;
    LinklistNode* prev = head;
    prev->m_Next = NULL;
    LinklistNode* cur = head->m_Next;
    while (cur)
    {
        LinklistNode* next = cur->m_Next;
        cur->m_Next = prev;
        next->m_Next = cur;
        prev = cur;
        cur = next;
    }
    return prev;
}
```
